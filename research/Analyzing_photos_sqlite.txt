## Total photos and videos ##

select count(Z_PK) from ZGENERICASSET;


## Total photos ##
select count(Z_PK) from ZGENERICASSET where ZKIND=0;

## Total videos ##
select count(Z_PK) from ZGENERICASSET where ZKIND=1;

## select relative path ##
select ('Media/' || ZDIRECTORY || '/' || ZFILENAME) as RelativePath from ZGENERICASSET;

select ('Media/' || ZDIRECTORY || '/' || ZFILENAME) as RelativePath from ZGENERICASSET where ZKIND=1;
